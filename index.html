        void Awake()
        {
            GameObject mainBase = GameObject.FindGameObjectWithTag("MainBase");
            if (mainBase != null)
            {
                mainBasePos = mainBase.transform.position;
                var depots = GameObject.FindGameObjectsWithTag("StorageDepot");
                depotPositions = new Vector3[depots.Length];
                for (int i = 0; i < depots.Length; i++)
                    depotPositions[i] = depots[i].transform.position;
            }
            else
            {
                Debug.LogError("ResourceSpawner2: MainBase not found.");
            }
        }

async function populateDropdowns() {
  const res = await fetch('/categories/');
  const { data: categories } = await res.json();
  const catSel = document.getElementById('category_id');
  catSel.innerHTML = '<option value="">Select a category</option>';
  categories.forEach(c => {
    const o = document.createElement('option');
    o.value = c.id;
    o.textContent = c.name;
    catSel.appendChild(o);
  });
}
document.addEventListener('DOMContentLoaded', populateDropdowns);
